# Generated by Poetica (target: python)

def obstacle_stop():
    sensor_distance = 25
    motor_speed = 100
    if sensor_distance < 10:
        motor_speed = 0
        print("obstacle detected")
    else:
        print("path clear")
    print(sensor_distance)
    return "check complete"


if __name__ == "__main__":
    obstacle_stop()
